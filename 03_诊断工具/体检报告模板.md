## 一、报告整体设计

### 1.1 设计原则

**清晰性**：
- 结论前置，关键数字突出
- 层次分明，逐级展开
- 避免术语，通俗易懂

**可操作性**：
- 不仅告知现状，更给出方案
- 每个缺口都有对应改造方案
- 明确下一步行动

**专业性**：
- 数据可视化（图表、进度条）
- 合规表述（无承诺）
- 品牌一致性

### 2.4 文案模板

**诊断建议生成逻辑**：

```javascript
function generateDiagnosisSummary(data) {
  const maxAmount = data.current_status.max_amount;
  const productCount = data.current_status.product_count;
  const potentialIncrease = data.summary.potential_increase;
  
  let summary = "";
  
  // 1. 当前状态评价
  if (productCount >= 5) {
    summary += "您当前可以申请中高额度产品，";
  } else if (productCount >= 2) {
    summary += "您当前有部分产品可选择，";
  } else if (productCount >= 1) {
    summary += "您当前可以申请少数产品，";
  } else {
    summary += "您当前暂时没有完全匹配的产品，";
  }
  
  // 2. 推荐产品
  if (productCount >= 1) {
    const topProduct = data.current_status.recommended_products[0];
    summary += `建议优先申请${topProduct.product_name}（额度最高）。`;
  }
  
  // 3. 改造建议
  if (potentialIncrease > 0) {
    const topScheme = data.recommended_schemes[0];
    summary += `通过${topScheme.scheme_name}，可进一步增加额度。`;
  } else {
    summary += "您的条件已经很优秀，可以直接申请。";
  }
  
  return summary;
}
```

import jsPDF from 'jspdf';
import 'jspdf-autotable';

function generatePDF(reportData) {
  const doc = new jsPDF();
  
  // 第1页：封面
  doc.setFontSize(24);
  doc.text('企业贷款体检报告', 105, 80, { align: 'center' });
  doc.setFontSize(16);
  doc.text(reportData.company_name || '企业名称', 105, 100, { align: 'center' });
  doc.setFontSize(12);
  doc.text(`生成日期：${reportData.generate_date}`, 105, 110, { align: 'center' });
  
  // 第2页：体检结论
  doc.addPage();
  doc.setFontSize(18);
  doc.text('🎯 体检结论', 20, 30);
  doc.setFontSize(32);
  doc.text(`${reportData.max_amount} 万元`, 105, 60, { align: 'center' });
  doc.setFontSize(12);
  doc.text(`✅ 匹配产品：${reportData.product_count} 个`, 20, 80);
  doc.text(`📈 潜在提升：+${reportData.potential_increase} 万元`, 20, 90);
  
  // 第3页：产品列表（使用autoTable）
  doc.addPage();
  doc.autoTable({
    head: [['产品名称', '额度', '利率', '期限']],
    body: reportData.products.map(p => [
      p.product_name,
      `${p.amount}万`,
      `${p.rate_min}%-${p.rate_max}%`,
      p.duration
    ]),
    startY: 30
  });
  
  // ...其他页面
  
  // 保存PDF
  doc.save(`贷款体检报告-${reportData.company_name}-${reportData.generate_date}.pdf`);
}
```

// 数字递增动画
function animateNumber(element, targetValue, duration = 1000) {
  let startValue = 0;
  const startTime = Date.now();
  
  function update() {
    const currentTime = Date.now();
    const elapsed = currentTime - startTime;
    const progress = Math.min(elapsed / duration, 1);
    
    const currentValue = Math.floor(startValue + (targetValue - startValue) * progress);
    element.textContent = currentValue;
    
    if (progress < 1) {
      requestAnimationFrame(update);
    }
  }
  
  update();
}
```
